Install full ffmpeg Homebrew

with tags ffmpeg - Saturday, November 3, 2012

Will install ffmpeg with all the codecs that I care about (namely fdk-aac).

brew install ffmpeg \
  --with-fdk-aac \
  --with-ffplay \
  --with-freetype \
  --with-frei0r \
  --with-libass \
  --with-libcaca \
  --with-libvo-aacenc \
  --with-libvorbis \
  --with-libvpx \
  --with-opencore-amr \
  --with-openjpeg \
  --with-openssl \
  --with-opus \
  --with-rtmpdump \
  --with-schroedinger \
  --with-speex \
  --with-theora \
  --with-tools
